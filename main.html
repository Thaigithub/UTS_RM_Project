def option_price(S, K, T, r, sigma, option_type):
    d1 = (np.log(S / K) + (r + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
    d2 = d1 - sigma * np.sqrt(T)
    if option_type == "Call":
        return S * norm.cdf(d1) - K * np.exp(-r * T) * norm.cdf(d2)
    elif option_type == "Put":
        return K * np.exp(-r * T) * norm.cdf(-d2) - S * norm.cdf(-d1)


def delta(S, K, T, r, sigma, option_type):
    d1 = (np.log(S / K) + (r + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
    if option_type == "Call":
        return norm.cdf(d1)
    elif option_type == "Put":
        return norm.cdf(d1) - 1


def gamma(S, K, T, r, sigma):
    d1 = (np.log(S / K) + (r + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
    return norm.pdf(d1) / (S * sigma * np.sqrt(T))


def weight_calc(portfolio):
    weights = np.zeros(4)
    for i in range(len(portfolio)):
        S = asset.loc[portfolio.iloc[i, 1], "Spot"]
        K = portfolio.iloc[i, 3]
        T = tau
        sigma = asset.loc[portfolio.iloc[i, 1], "Volatility"]
        option_type = portfolio.iloc[i, 0]
        position = portfolio.iloc[i, 2]
        d = -delta(S, K, T, r, sigma, option_type) * position * S
        g = -0.5 * (
            gamma(S, K, T, r, sigma) * position * (S**2)
            + delta(S, K, T, r, sigma, option_type) * position * S
        )
        if portfolio.iloc[i, 1] == "CBA":
            weights[0] += d
            weights[2] += g
        else:
            weights[1] += d
            weights[3] += g
    return weights

def Method_1(portfolio):
    weights = weight_calc(portfolio)[:2]

    def info_cal(day, isOverlapping=None):
        risk_factor = None
        if day == 1 or isOverlapping:
            risk_factor = (
                log_ret.rolling(window=day).sum().dropna().reset_index(drop=True)
            )
        else:
            num_complete_groups = len(log_ret) // day
            cut = len(log_ret) - num_complete_groups * day
            risk_factor = (
                log_ret[cut:].groupby((log_ret[cut:].index - cut) // day).sum()
            )
        mean_distribution = weights @ risk_factor.mean().T
        var_distribution = weights @ risk_factor.cov().T @ weights.T
        std_distribution = np.sqrt(var_distribution)
        return mean_distribution, std_distribution

    def VaR(alpha, day, isScale=None, isOverlapping=None):
        if day == 1 or isScale:
            mean_distribution, std_distribution = info_cal(1)
            return mean_distribution * day + std_distribution * norm.ppf(
                alpha
            ) * np.sqrt(day)
        else:
            mean_distribution, std_distribution = info_cal(day, isOverlapping)
            return mean_distribution + std_distribution * norm.ppf(alpha)

    def ES(alpha, day, isScale=None, isOverlapping=None):
        if day == 1 or isScale:
            mean_distribution, std_distribution = info_cal(1)
            return mean_distribution * day + std_distribution * norm.pdf(
                norm.ppf(alpha)
            ) / (1 - alpha) * np.sqrt(day)
        else:
            mean_distribution, std_distribution = info_cal(day, isOverlapping)
            return mean_distribution + std_distribution * norm.pdf(norm.ppf(alpha)) / (
                1 - alpha
            )

    alphas = [0.9, 0.95, 0.99]
    ret_df = pd.DataFrame(
        {
            "1 day VaR": [VaR(alpha, 1) for alpha in alphas],
            "10 day VaR Scale": [VaR(alpha, 10, True) for alpha in alphas],
            "10 day VaR Overlapping": [VaR(alpha, 10, False, True) for alpha in alphas],
            "10 day VaR Non-Overlapping": [
                VaR(alpha, 10, False, False) for alpha in alphas
            ],
            "1 day ES": [ES(alpha, 1) for alpha in alphas],
            "10 day ES Scale": [ES(alpha, 10, True) for alpha in alphas],
            "10 day ES Overlapping": [ES(alpha, 10, False, True) for alpha in alphas],
            "10 day ES Non-Overlapping": [
                ES(alpha, 10, False, False) for alpha in alphas
            ],
        }
    )
    ret_df.index = [f"{alpha*100}%" for alpha in alphas]
    return ret_df

def Method_2(portfolio):
    weights = weight_calc(portfolio)

    def loss_cal(day):
        risk_factor = log_ret.rolling(window=day).sum().dropna().reset_index(drop=True)
        risk_factor = pd.concat([risk_factor, risk_factor**2], axis=1)
        return np.sort((weights @ risk_factor.T).to_numpy())

    def VaR(alpha, day):
        loss = loss_cal(day)
        return np.quantile(loss, alpha, method="higher")

    def ES(alpha, day):
        loss = loss_cal(day)
        var = VaR(alpha, day)
        return np.maximum(loss - var, 0).mean() / (1 - alpha) + var

    alphas = [0.9, 0.95, 0.99]
    ret_df = pd.DataFrame(
        {
            "1 day VaR": [VaR(alpha, 1) for alpha in alphas],
            "10 day VaR": [VaR(alpha, 10) for alpha in alphas],
            "1 day ES": [ES(alpha, 1) for alpha in alphas],
            "10 day ES": [ES(alpha, 10) for alpha in alphas],

def Method_3(portfolio, lambda_decay):
    weights = weight_calc(portfolio)

    def loss_cal(day):
        risk_factor = log_ret.rolling(window=day).sum().dropna().reset_index(drop=True)
        risk_factor = pd.concat([risk_factor, risk_factor**2], axis=1)
        loss = (weights @ risk_factor.T).to_numpy()
        hist_weight = (
            lambda_decay ** np.arange(len(loss))[::-1]
            * (1 - lambda_decay)
            / (1 - lambda_decay ** len(loss))
        )
        loss_weighted = np.column_stack((loss, hist_weight))
        loss_weighted = loss_weighted[loss_weighted[:, 0].argsort()]
        cumulative_weight = np.cumsum(loss_weighted[:, 1])
        loss_weighted = np.column_stack((loss_weighted, cumulative_weight))
        return loss_weighted

    def VaR(alpha, day):
        loss_weighted = loss_cal(day)
        loss_exceed = loss_weighted[(loss_weighted[:, 2] >= alpha)]
        return min(loss_exceed[:, 0])

    def ES(alpha, day):
        loss_weighted = loss_cal(day)
        loss_exceed = loss_weighted[(loss_weighted[:, 2] >= alpha)]
        var = VaR(alpha, day)
        return sum((loss_exceed[:, 0] - var) * loss_exceed[:, 1]) / (1 - alpha) + var

    alphas = [0.9, 0.95, 0.99]
    ret_df = pd.DataFrame(
        {
            "1 day VaR": [VaR(alpha, 1) for alpha in alphas],
            "10 day VaR": [VaR(alpha, 10) for alpha in alphas],
            "1 day ES": [ES(alpha, 1) for alpha in alphas],
            "10 day ES": [ES(alpha, 10) for alpha in alphas],
        }
    )
    ret_df.index = [f"{alpha*100}%" for alpha in alphas]
    return ret_df

def Method_5(portfolio, n):
    weights = weight_calc(portfolio)

    def loss_cal(day, isScale=None, isOverlapping=None):
        np.random.seed(0)
        risk_factor = None
        if day == 1 or isScale:
            risk_factor = (
                log_ret.rolling(window=1).sum().dropna().reset_index(drop=True)
            )
            std = risk_factor.std().to_numpy()
            mean = risk_factor.mean().to_numpy()
            risk_factor = (risk_factor - risk_factor.mean()) / risk_factor.std()
            [rho, degree], _ = estimation.fit_cmle(student(), risk_factor.values.T)
            cdf = simulation.simu_tstudent(
                2, n, np.array([[1, rho], [rho, 1]]), degree
            ).T
            risk_factor = t.ppf(cdf, degree) * std * np.sqrt(day) + mean * day
            risk_factor = np.hstack([risk_factor, risk_factor**2])
        else:
            if isOverlapping:
                risk_factor = (
                    log_ret.rolling(window=1).sum().dropna().reset_index(drop=True)
                )
            else:
                num_complete_groups = len(log_ret) // day
                cut = len(log_ret) - num_complete_groups * day
                risk_factor = (
                    log_ret[cut:].groupby((log_ret[cut:].index - cut) // day).sum()
                )
            std = risk_factor.std().to_numpy()
            mean = risk_factor.mean().to_numpy()
            risk_factor = (risk_factor - risk_factor.mean()) / risk_factor.std()
            [rho, degree], _ = estimation.fit_cmle(student(), risk_factor.values.T)
            cdf = simulation.simu_tstudent(
                2, n, np.array([[1, rho], [rho, 1]]), degree
            ).T
            risk_factor = t.ppf(cdf, degree) * std + mean
            risk_factor = np.hstack([risk_factor, risk_factor**2])

    def VaR(alpha, day, isScale=None, isOverlapping=None):
